const BLOBS = Array.from({length:7},(_,i)=>({
  x:Math.random()*800, y:Math.random()*800,
  vx:(Math.random()-.5)*.18, vy:(Math.random()-.5)*.18,
  r:120+Math.random()*200,
  phase:Math.random()*Math.PI*2,
  speed:0.004+Math.random()*0.006
}));

const TENDRILS = Array.from({length:12},()=>({
  pts:[],
  x:Math.random()*800, y:Math.random()*900,
  angle:Math.random()*Math.PI*2,
  speed:0.003+Math.random()*0.005,
  length:60+Math.random()*120,
  width:2+Math.random()*5,
  alpha:0.06+Math.random()*0.1,
  wobble:Math.random()*Math.PI*2
}));

let gt=0;
function drawGoo(){
  gt++;
  gctx.clearRect(0,0,GW,GH);
  for(const b of BLOBS){
    b.phase+=b.speed;
    b.x+=b.vx+Math.sin(b.phase*0.7)*0.3;
    b.y+=b.vy+Math.cos(b.phase*0.5)*0.3;
    if(b.x<-b.r)b.x=GW+b.r;
    if(b.x>GW+b.r)b.x=-b.r;
    if(b.y<-b.r)b.y=GH+b.r;
    if(b.y>GH+b.r)b.y=-b.r;
    const pulse=0.85+Math.sin(b.phase)*0.15;
    const rr=b.r*pulse;
    const g=gctx.createRadialGradient(b.x,b.y,0,b.x,b.y,rr);
    g.addColorStop(0,'rgba(0,0,0,0.55)');
    g.addColorStop(0.5,'rgba(0,0,0,0.25)');
    g.addColorStop(1,'rgba(0,0,0,0)');
    gctx.fillStyle=g;
    gctx.beginPath();gctx.arc(b.x,b.y,rr,0,Math.PI*2);gctx.fill();
  }
  for(const td of TENDRILS){
    td.wobble+=td.speed;
    td.angle+=Math.sin(td.wobble)*0.008;
    const pts=[];
    let cx=td.x,cy=td.y;
    const segs=12;
    for(let i=0;i<segs;i++){
      const t2=i/segs;
      const wx=Math.sin(td.wobble+t2*3)*20*(1-t2);
      pts.push([cx+wx,cy]);
      cy+=td.length/segs;
      cx+=Math.sin(td.wobble*0.5+i)*3;
    }
    const tip=pts[pts.length-1];
    const blobR=(td.width*1.8)*(0.7+Math.sin(td.wobble*2)*0.3);
    const bg=gctx.createRadialGradient(tip[0],tip[1],0,tip[0],tip[1],blobR*2);
    bg.addColorStop(0,`rgba(0,0,0,${td.alpha*1.5})`);
    bg.addColorStop(1,'rgba(0,0,0,0)');
    gctx.fillStyle=bg;
    gctx.beginPath();gctx.arc(tip[0],tip[1],blobR*2,0,Math.PI*2);gctx.fill();
    gctx.beginPath();
    gctx.moveTo(pts[0][0],pts[0][1]);
    for(let i=1;i<pts.length;i++){
      const [px,py]=pts[i-1],[nx,ny]=pts[i];
      gctx.quadraticCurveTo(px,py,(px+nx)/2,(py+ny)/2);
    }
    for(let i=0;i<pts.length-1;i++){
      const t2=i/(pts.length-1);
      const w=td.width*(1-t2*0.7);
      gctx.lineWidth=w;
      gctx.strokeStyle=`rgba(0,0,0,${td.alpha*(1-t2*0.3)})`;
      gctx.beginPath();
      gctx.moveTo(pts[i][0],pts[i][1]);
      gctx.lineTo(pts[i+1][0],pts[i+1][1]);
      gctx.stroke();
    }
    td.x+=Math.sin(td.wobble*0.3)*0.4;
    td.y+=Math.cos(td.wobble*0.2)*0.15;
    if(td.x<-50)td.x=GW+50;
    if(td.x>GW+50)td.x=-50;
    if(td.y<-td.length)td.y=GH;
    if(td.y>GH+td.length)td.y=-td.length;
  }
  requestAnimationFrame(drawGoo);
}
drawGoo();
